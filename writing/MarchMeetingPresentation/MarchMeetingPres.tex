                    (\text{Predation}) & X_nX_{n+1} \rightarrow{\sigma} X_n \varnothing \\

                    (\text{Hopping}) & XY \rightarrow{\epsilon_r} YX \\
                \end{align*}
                Forms coherent spiral waves (with certain parameters).
        \end{itemize}
